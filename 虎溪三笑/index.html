  <script>

    const correctAnswers = ['card-tao', 'card-hui', 'card-lu'];

    let currentAnswers = [null, null, null];


    const thinkers = [

      { id: 'card-tao', name: '陶淵明' },

      { id: 'card-hui', name: '慧遠' },

      { id: 'card-lu', name: '陸修靜' },

      { id: 'card-confucius', name: '韓非子' },

      { id: 'card-laozi', name: '楊朱' },

      { id: 'card-zhuangzi', name: '鄒衍' }

    ];


    // Shuffle the thinkers array

    thinkers.sort(() => Math.random() - 0.5);


    // Render shuffled cards

    const thinkerCardsContainer = document.getElementById('thinker-cards');

    thinkers.forEach(thinker => {

      const card = document.createElement('div');

      card.className = 'thinker-card';

      card.id = thinker.id;

      card.draggable = true;

      card.innerHTML = `

        <div class="thinker-image-placeholder"></div>

        <div>${thinker.name}</div>

      `;

      card.addEventListener('dragstart', (e) => {

        e.dataTransfer.setData('text/plain', e.target.id);

      });

      thinkerCardsContainer.appendChild(card);

    });


    const slots = document.querySelectorAll('.password-slot');

    const submitButton = document.getElementById('submit-button');

    const resetButton = document.getElementById('reset-button');

    const resultMessage = document.getElementById('result-message');


    slots.forEach((slot, index) => {

      slot.addEventListener('dragover', (e) => {

        e.preventDefault();

      });

      slot.addEventListener('drop', (e) => {

        e.preventDefault();

        const cardId = e.dataTransfer.getData('text/plain');

        const cardElement = document.getElementById(cardId);

        if (slot.firstChild) {

          thinkerCardsContainer.appendChild(slot.firstChild);

        }

        slot.innerHTML = "";

        slot.appendChild(cardElement);

        currentAnswers[index] = cardId;

      });

    });


    submitButton.addEventListener('click', () => {

      if (JSON.stringify(correctAnswers.sort()) === JSON.stringify(currentAnswers.filter(Boolean).sort())) {

        resultMessage.style.color = 'green';

        resultMessage.textContent = '智慧值 +10！';

        resetButton.style.display = 'none';

      } else {

        resetButton.style.display = 'inline-block';

      }

    });


    resetButton.addEventListener('click', resetGame);


    function resetGame() {

      resultMessage.textContent = '';

      resetButton.style.display = 'none';

      currentAnswers = [null, null, null];

      slots.forEach(slot => {

        if (slot.firstChild) {

          thinkerCardsContainer.appendChild(slot.firstChild);

        }

      });

      slots.forEach(slot => slot.innerHTML = '');

    }

  </script>
